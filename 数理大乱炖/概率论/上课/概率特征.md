$$
\begin{gathered}
E(X)=\int x f(x) d x \\
E(g(X))=\int g(x) f(x) d x \\
E(a X+b)=a E(X)+b
\end{gathered}
$$

\begin{gathered}
\operatorname{Var}(X)=E\left(X^2\right)-[E(X)]^2 \\
\operatorname{Var}(a X+b)=a^2 \operatorname{Var}(X)
\end{gathered}
$$
